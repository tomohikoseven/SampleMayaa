処理をまとめる(m:with)

withプロセッサを使っています。

with.html 

<html>
<head>
<meta http-equiv="Content-Type" content="text/html; charset=UTF-8" />
<title>処理をまとめる(m:with)</title>
</head>
<body>
    <h3>処理をまとめる(m:with)</h3>
    <p><span id="with">withプロセッサを使っています。</span></p>
</body>
</html>
with.mayaa 
        
<?xml version="1.0" encoding="UTF-8"?>
<m:mayaa xmlns:m="http://mayaa.seasar.org">
    <m:with m:id="with">
        <m:if test="${randNum == 1}">
            <m:write value="randNumは１です。" />
        </m:if>
        <m:if test="${randNum != 1}">
            <m:write value="randNumは１ではありません。" />
        </m:if>
    </m:with>
</m:mayaa>
WithAction.java 
        
package sampleMayaa.action;

import java.util.Random;
import org.seasar.struts.annotation.Execute;

public class WithAction {

    public int randNum;

    @Execute(validator = false)
    public String index() {
        Random rand = new Random();
        randNum = rand.nextInt(2);  // 0と1をランダムに生成
        return "with.html";
    }
}
