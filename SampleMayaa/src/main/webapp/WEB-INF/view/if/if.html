条件分岐(m:if)

1を出力しています。

0を出力しています。

 if.html 

<html>
<head>
<meta http-equiv="Content-Type" content="text/html; charset=UTF-8" />
<title>条件分岐(m:if)</title>
</head>
<body>
    <h3>表示</h3>
    <span id="if_1"><p><span id="if_1_message">1を出力しています。</span></p></span>
    <span id="if_0"><p><span id="if_0_message">0を出力しています。</span></p></span>
</body>
</html>
if.mayaa 
        
<?xml version="1.0" encoding="UTF-8"?>
<m:mayaa xmlns:m="http://mayaa.seasar.org">
    <m:if m:id="if_1" test="${randNum == 1 }"/>
    <m:if m:id="if_0" test="${randNum != 1 }"/>
    <m:write m:id="if_1_message" value="1を出力。"/>
    <m:write m:id="if_0_message" value="0を出力。"/>
</m:mayaa>
IfAction.java 
        
package sampleMayaa.action;

import java.util.Random;
import javax.annotation.Resource;
import org.seasar.struts.annotation.Execute;

public class IfAction {

    public int randNum;

    @Execute(validator = false)
    public String index() {
        Random rand = new Random();
        randNum = rand.nextInt(2);  // 0と1をランダムに生成
        return "if.html";
    }
}
